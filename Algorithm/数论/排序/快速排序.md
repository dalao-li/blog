##  快速排序

```c
// left，reight代表数组左右区间大小
template<class T>
void quickSort(vector<T> &v，int left，int right) {
	// index表示基准下标
    // 通过调换使得index左侧值均比v[index]小，右侧值均比它大
	int index = left;
	// 设定基准元素，默认基准位置为最左边位置
	int pivot = v[left];
	for(int i = left + 1 ; i <= right ; i++) {
		if(v[i] < pivot) {
			// 基准下标右移
			index++;
			// 如果待交换位置与该位置不重合
			if(index != i) {
				swap(v[i]，v[index]);
			}
		}
	}
	// 基准元素就位，基准位为index
	v[left] = v[index];
	v[index] = pivot;
	if(left < right) {
		// 对左，右侧元素也进行同样处理
		quicksort(v，left，index-1);
		quicksort(v，index+1 ，right);
	}
}
```
