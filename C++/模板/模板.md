### 结构体模板


```c++
template<class T>
struct Node {
    T data;
    Node<T> *next;
};

template<class T>
void display(Node<T> *&head, vector<T> &v) {
    Node<T> *tail = head;

    for(int i = 0; i < v.size(); i++) {
        Node<T> *p = new Node<T>;

        p->data = v[i];
        p->next = NULL;

        // 新节点位于尾指针的后边
        tail->next = p;
        // 更新尾指针的位置
        tail = p;
    }

    Node<T> *p = head->next;
    while(p) {
        std::cout << p->data << " ";
        p = p->next;
    }
}
```



### 类模板


```c++
template <class T>
class Composer {
public:
    Composer(T a, T b) {
        x = a;
        y = b;
    }

    T max();
    T min();
    void display();

private:
    T x;
    T y;
};

template<class T>
T Composer<T>::max() {
    return x > y ? x : y;
}

template<class T>
T Composer<T>::min() {
    return x < y ? x : y;
}

template<class T>
void Composer<T>::display() {
    std::cout << max() << " " << min() << std::endl;

}
```
