```c
#include<iostream>
#include<fstream>
#include<cstdio>
using namespace std;

int main(void) {
    string str;
    ifstream in("Example.txt" , ios::in);

    ofstream out("Example.txt" , ios::in);
    
    if(!in) {
        cerr << "open error!" << endl;
        exit(1);
    }
    while(getline(in , str)) {
        out << str;
        out << str << endl;
    }
    cout << endl;
    in.close();
    return 0;
}
```
