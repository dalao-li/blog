```c++
#include<iostream>
#include<string>

using namespace std;

class People {
    public:
        People(){}
        People(int a , string s){
            age = a;
            name = s;
        }
        void setValue(int age , string name){
            this.age = age;
            this.name = name;
        } 
        virtual void display() = 0;
        ~People(){}
    
    protected:
        int age;
        string name;
};


class Student : public People {
    public:
        Student(){}
        Student(int a , string s , int b):People(a , s),ID(b){}
        void setID(string ID){
            this.ID = ID;
        }
        void display(){
            printf("age = %d\nname = %s\nID = %d\n" , age , name , ID);
        }
        ~Student(){}
    
    private:
        int ID;
}; 

class Teacher:public People {
    public:
        Teacher(){}
        Teacher(int a , string s , int b):People(a , s),ID(b){}
        void setID(){
            this.ID = ID;
        }
        void display(){
            printf("age = %d\nname = %s\nID = %d\n" , age , name , ID);
        }
        ~Teacher(){}
    
    private:
        int ID;
};

int main()
{
    Student *p1 , stu;
    Teacher *p2 , tea;
    p1 = &stu;
    p1->setValue(20 , 'LILI');
    p1->setID(1009);
    p1->display();
    
    p2 = &tea;
    p2->setValue(30 , 'WANG');
    p2->setID(2001);
    p2->display();
    return 0;
}
```
