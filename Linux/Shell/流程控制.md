## if

```sh
if [ command ];then
   ...
elif [ command ];then
   ...
else
   ...
fi
```

## case

```sh
case 值 in
模式1)
    ...
    ;;
模式2）
    ...
    ;;
esac
```

## for

```sh
# 例，累计循环
for i in {1..5};do
    echo $i
done
```

```sh
# 例，遍历
for i in This is a string;do
    echo ${i}
done

# 例，遍历文件
for cfg_file in grub.cfg anaconda-ks.cfg default;do
   sed -i "s#DEFAULT_IP#$SERVER_IP#g" /root/many_ks/${cfg_file} 
done
```

## while

```sh
i=1

while(( ${i}<=5 ));do
    echo ${i}
    let "int++" 
done
```
