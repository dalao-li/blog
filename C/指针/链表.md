- 定义

```c++
typedef struct Node {
	int data;
	struct Node *next;
} Node，*NodeList;
```

- 初始化

```c++
NodeList init(){
    NodeList head = (NodeList)malloc(sizeof(Node));
	head->next = NULL;
    return head;
}
```

- 输入

```c++
//输入n个节点的值
void input(NodeList &head，int n){
	//起始头指针与尾指针在一起
	NodeList tail = head;
	NodeList p;
	for(int i = 0; i < n; i++) {
        //创建新节点
	    p = (NodeList)malloc(sizeof(Node));
	    scanf("%d"，&p->data);
	    p->next = NULL;
        //新节点位于尾指针的后边
		tail->next = p;
        //更新尾指针的位置
		tail = p;
	}
}
```

- 输出

```c++
void ouput(NodeList &head){
    NodeList p = head->next;
    while(p){
        printf("%d "，p->data);
        p = p->next;
    }
}
```
